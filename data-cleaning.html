3.2 Cleaning the Transcript Dataset

# Updated 11/30/2023
filtered_cohort_transcripts <- unfiltered %>%
  mutate(RS5TERM = ifelse(INSTNM == "Environmental Studies", 3, RS5TERM)) %>% # Looking for study abroad courses run by the ES department and putting them into the appropriate semester
  filter(RS5TERM %in% c(1, 2, 3), # Filtering to only include classes in Fall, January, Spring semesters
         RS5CREDS ==1, # Filtering to only include 1 credit courses
         INSTNM %in% c("St. Olaf College", "Environmental Studies")) %>% # Filtering to only include courses offered through St. Olaf
  mutate(ES_course = ifelse(RL1CSDAB %in% c("ENVST", "AS/ES", "BI/ES", "ES/PS") | (RL1CSDAB == "ENCON" & RL1CSNUM == "111"), TRUE, FALSE)) %>% # Creating a new variable to indicate if a course is from an ES department
  rename("CohortYear" = GS1COYR,
         "Year" = RS5YEAR,
         "Term" = RS5TERM,
         "Department" = RL1CSDAB,
         "Course_Number" = RL1CSNUM,
         "Course_Name" = RL1CSNM,
         "Credits" = RS5CREDS,
         "Grade" = RS5GRCD) %>% # Renaming to readable variable names
  select(-CLBID, -RL1INST, -RL1HALF, -Credits, -INSTNM) %>% # Remove unneeded variables
  mutate(Year_Taken = Year-CohortYear,
         Total_Term = (Year_Taken*3)+Term) # Creating a variable to indicate what term a course is taken in, where terms 1-3 are in the first year, 4-6 are in sophomore year, etc.

# Tibble for filtering out non-four year students
filter_out <- filtered_cohort_transcripts %>%
  filter(Total_Term > 12) %>%
  group_by(PPNUM) %>%
  summarize()

# Filtering out non-four year students
filtered_cohort_transcripts <- filtered_cohort_transcripts %>%
  anti_join(filter_out)

# Write cleaned version of `unfiltered` tibble/dataset
write_csv(filtered_cohort_transcripts, "/home/rstudio/users/kziegler/ENVST CIR 2023_24/Data/CSV Data/filtered_cohort_transcripts.csv")

# Updated 11/30/2023
# Converting dates to terms
envst_maj_emph_conc_all_records_clean <- envst_maj_emph_conc_all_records %>%
  mutate(Date = ymd(`RS2CHGDT or RS3CHGDT`), # reading the column with declaration/withdrawal dates as a date 
         Year = as.numeric(year(Date))-GS1COYR, # Finding the year category that corresponds with each declaration/withdrawal event (years 1, 2, 3, 4 of college)
         month = month(Date), # extracting the month of declaration/withdrawal dates
         Term = ifelse(month == 1, 2, ifelse((month %in% c(2, 3, 4, 5, 6, 7, 8)), 3, ifelse(month %in% c(9, 10, 11, 12), 1, 0))), # Sorting the declaration/withdrawal dates into terms - if delcaration/withdrawal occured in months 2-8, the date is recorded as part of Spring semester. If it occurred during months 9-12, it is recorded as part of Fall semester.
         Year = ifelse(Term %in% c(2, 3), Year-1, Year), # Year is adjusted from calender year to school year
         Total_Term = (Year*3)+Term) # The date of declaration/withdrawal is converted into total term, on the same scale as the courses taken

#Finding concentration declarations
